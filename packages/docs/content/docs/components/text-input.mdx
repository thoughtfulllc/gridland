## Installation

```bash title="Terminal"
bunx create-gridland add text-input
```

## Usage

```tsx
import { TextInput } from "@/components/ui/text-input"
```

```tsx
const [name, setName] = useState("")

<TextInput
  label="Username"
  value={name}
  onChange={setName}
  placeholder="enter your name"
  prompt="> "
/>
```

### With onSubmit

Use `onSubmit` to handle Enter key presses.

```tsx title="With onSubmit"
const [email, setEmail] = useState("")

<TextInput
  label="Email"
  value={email}
  onChange={setEmail}
  onSubmit={(v) => console.log("Submitted:", v)}
  placeholder="user@example.com"
  prompt="> "
/>
```

### Required

Show a required indicator (`*`) next to the label.

```tsx title="Required"
const [name, setName] = useState("")

<TextInput
  label="Name"
  value={name}
  onChange={setName}
  placeholder="enter your name"
  prompt="> "
  required
/>
```

### Description

Display helper text inline next to the label.

```tsx title="With description"
const [email, setEmail] = useState("")

<TextInput
  label="Email"
  value={email}
  onChange={setEmail}
  placeholder="user@example.com"
  prompt="> "
  description="We'll never share your email"
/>
```

### Error

Pass an `error` string to show a validation message. It replaces the description.

```tsx title="With error"
const [email, setEmail] = useState("")

<TextInput
  label="Email"
  value={email}
  onChange={setEmail}
  placeholder="user@example.com"
  prompt="> "
  error="This field is required"
/>
```

### Max Length

Limit input length and show a character counter when the user starts typing.

```tsx title="Max length"
const [bio, setBio] = useState("")

<TextInput
  label="Bio"
  value={bio}
  onChange={setBio}
  placeholder="tell us about yourself"
  prompt="> "
  maxLength={100}
/>
```

### Disabled

A disabled input ignores focus and keystrokes.

```tsx title="Disabled"
<TextInput
  label="API Key"
  value="sk-1234...abcd"
  placeholder="sk-..."
  prompt="> "
  disabled
/>
```

### Form with keyboard navigation

TextInput is a controlled primitive — it takes a `focus` prop rather than managing focus internally. Compose multiple inputs into a form by tracking the active field index and wiring arrow keys.

```tsx title="Form"
import { useKeyboard } from "@gridland/utils"

const FIELDS = ["Username", "Email", "Password"]

function MyForm() {
  const [active, setActive] = useState(0)
  const [values, setValues] = useState(FIELDS.map(() => ""))

  useKeyboard((event) => {
    if (event.name === "up") setActive((i) => Math.max(0, i - 1))
    if (event.name === "down") setActive((i) => Math.min(FIELDS.length - 1, i + 1))
  })

  return (
    <>
      {FIELDS.map((label, i) => (
        <TextInput
          key={label}
          label={label}
          value={values[i]}
          onChange={(v) => setValues((prev) => prev.map((old, j) => j === i ? v : old))}
          focus={i === active}
          prompt="> "
        />
      ))}
    </>
  )
}
```

## Controls

- **Enter**: Submit the current value
- **↑/↓**: Navigate between fields (when composed in a form)

## API Reference

### TextInput

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | - | Field label shown above the input |
| `description` | `string` | - | Helper text shown inline next to the label |
| `error` | `string` | - | Error message — overrides description when set |
| `required` | `boolean` | `false` | Show required indicator (`*`) on the label |
| `disabled` | `boolean` | `false` | Disable the input |
| `value` | `string` | *required* | Current value of the input |
| `onChange` | `(value: string) => void` | - | Called on every keystroke |
| `onSubmit` | `(value: string) => void` | - | Called when Enter is pressed |
| `placeholder` | `string` | - | Hint text shown when the input is empty |
| `prompt` | `string` | - | Prompt string before the input (e.g. `"> "`) |
| `focus` | `boolean` | `false` | Whether the input is focused and accepting keystrokes |
| `maxLength` | `number` | - | Maximum characters allowed (shows counter when typing) |
