## Configure the Vite Plugin

Polyterm provides a Vite plugin that handles module resolution for the
underlying opentui engine. Add it to your `vite.config.ts`:

```ts title="vite.config.ts"
import { defineConfig } from "vite"
import react from "@vitejs/plugin-react"
import { polytermWebPlugin } from "@polyterm.io/web/vite-plugin"

export default defineConfig({
  plugins: [
    react(),
    ...polytermWebPlugin(),
  ],
})
```

</Step>
<Step>

## Create Your Component

Wrap your TUI content in `TUI`:

```tsx title="App.tsx"
import { TUI } from "@polyterm.io/web"

function PolytermDemo() {
  return (
    <box flexDirection="column" padding={1}>
      <box border borderStyle="rounded" borderColor="#88c0d0" padding={1}>
        <text fg="#a3be8c">Hello from Vite!</text>
      </box>
    </box>
  )
}

export default function App() {
  return (
    <TUI style={{ width: "100vw", height: "100vh" }}>
      <PolytermDemo />
    </TUI>
  )
}
```
